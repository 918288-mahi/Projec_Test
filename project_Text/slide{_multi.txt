    <script>
        function mySlider(slider) {
            let currentIndex = 0;
            const innerContainer = slider.querySelector('.inner-container');
            const items = innerContainer.querySelectorAll('.slider');
            const len = items.length;
            const prevButton = slider.querySelector('.prev');
            const nextButton = slider.querySelector('.next');

            function updateButtons() {
                if (prevButton.classList.contains('disabled')) {
                    prevButton.disabled = currentIndex === 0;
                }
                if (nextButton.classList.contains('disabled')) {
                    nextButton.disabled = currentIndex === len - 1;
                }
            }

            function slideTo(index) {
                let position = -index * 100;
                innerContainer.style.transform = `translateX(${position}%)`;
                currentIndex = index;
                updateButtons();
            }

            prevButton.addEventListener('click', function() {
                if (currentIndex > 0) {
                    slideTo(currentIndex - 1);
                }
            });

            nextButton.addEventListener('click', function() {
                if (currentIndex < len - 1) {
                    slideTo(currentIndex + 1);
                }
            });

            updateButtons();
        }

        document.addEventListener('DOMContentLoaded', function() {
            const sliders = document.querySelectorAll('.main_slider_div');
            sliders.forEach(function(slider) {
                mySlider(slider);
            });
        });
    </script>
